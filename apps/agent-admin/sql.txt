-- 修补所有缺失的 channel_code
UPDATE PaymentRecords pr
LEFT JOIN Users u ON pr.user_id = u.id
SET pr.channel_code = u.channel_code
WHERE (pr.channel_code IS NULL OR pr.channel_code = '')
   AND u.channel_code IS NOT NULL 
   AND u.channel_code != '';

查看每个 代理的收入
SELECT 
    u.channel_code,
    pr.id as payment_id,
    pr.amount,
    pr.payment_way,
    pr.created_at,
    pr.payment_status,
    u.id as user_id
FROM Admins a
LEFT JOIN Users u ON JSON_CONTAINS(a.allowed_channel_codes, CONCAT('"', u.channel_code, '"'))
LEFT JOIN PaymentRecords pr ON u.id = pr.user_id 
    AND pr.payment_status = 3 
    AND (pr.payment_way NOT LIKE '%平台币%' OR pr.payment_way IS NULL OR pr.payment_way = '')
WHERE a.name = 'dao008' 
    AND a.is_active = 1
    AND pr.id IS NOT NULL  -- 只显示有支付记录的
ORDER BY pr.created_at DESC;

SELECT
      pr.sub_user_id,
      pr.world_id as server_id,
      COALESCE(SUM(pr.amount), 0) as total_amount,
      COUNT(*) as total_transactions,
      MIN(DATE(pr.created_at)) as first_consumption_date,
      MAX(DATE(pr.created_at)) as last_consumption_date
  FROM PaymentRecords pr
  WHERE pr.sub_user_id = 1003782
      AND pr.world_id = 3
      AND pr.payment_status IN (3, 4)
  GROUP BY pr.sub_user_id, pr.world_id;


  SELECT
      pr.sub_user_id,
      pr.world_id as server_id,
      DATE(pr.created_at) as consumption_date,
      COALESCE(SUM(pr.amount), 0) as daily_amount,
      COUNT(*) as daily_transactions
  FROM PaymentRecords pr
  WHERE pr.sub_user_id = 1003782
      AND pr.world_id = 3
      AND pr.payment_status IN (3, 4)
      AND DATE(pr.created_at) >= DATE_SUB(CURDATE(), INTERVAL 2 DAY)
  GROUP BY pr.sub_user_id, pr.world_id, DATE(pr.created_at)
  ORDER BY DATE(pr.created_at) DESC;
