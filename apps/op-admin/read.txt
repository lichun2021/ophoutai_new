
   下载并安装 NVM
   curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash
   # 使 NVM 立即生效
   source ~/.bashrc
   # 验证 NVM 安装
   nvm --version
   # 安装 Node.js 20
   nvm install 20
   # 设置为默认版本
   nvm alias default 20
   
   npm install pm2@latest -g
